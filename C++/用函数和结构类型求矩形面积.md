#include <iostream>
using namespace std;

struct RECT{
	float x;
	float y;
};

float area(RECT a1,RECT a2)
{
	/*if (a1.x > a2.x)

		}
		else{
			return (a1.x - a2.x)*(a2.y - a1.y);
		}
	}
	else{
		if (a1.y > a2.y)
		{
			return (a2.x - a1.x)*(a1.y - a2.y);

	int width = a1.x > a2.x ? a1.x - a2.x : a2.x - a1.x;
	int height = a1.y > a2.y ? a1.y - a2.y : a2.y - a1.y;
	return width*height;
}

void main()
{
	RECT sfu;
	RECT srd;
	cout << "请输入矩形左上角x坐标的值" << endl;
	cin >> sfu.x;
	cout << "请输入矩形左上角y坐标的值" << endl;
	cin >> sfu.y;
	cout << "请输入矩形右下角角x坐标的值" << endl;
	cin >> srd.x;
	cout << "请输入矩形右下角角y坐标的值" << endl;
	cin >> srd.y;
	cout << "矩形的面积为 " << area(sfu, srd) << endl;
	system("pause");
}
