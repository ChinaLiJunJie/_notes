# 用类和对象求圆面积
## 头文件
```
#include <iostream>
using namespace std;

class Point
{
private:
	int x;
	int y;
public:
	void cin (int x, int y);
	int getx();
	int gety();
};
class Circle
{
private:
	double PI = 3.14;
	double r;
	Point i;
public:
	double area();
	void inputR(double r);
	void outputpoint();
	void inputi(int x, int y);
};
```
## 执行文件
```
#include "learn.h"

void Point::cin(int x, int y)
{
	this->x = x;
	this->y = y;
}
double Circle::area()
{
	cout << "圆面积为" << PI*r*r << endl;
	return PI*r*r;
}
void Circle::inputR(double r)
{
	this->r = r;
}
void Circle::inputi(int x,int y)
{
	i.cin(x, y);
}
void Circle::outputpoint()
{
	cout << "圆心坐标为" << "(" << i.getx() << "." << i.gety() << ")" << endl;
}
int Point::getx()
{
	return x;
}
int Point::gety()
{
	return y;
}
void main()
{
	Circle s1;
	s1.inputR(5);
	s1.inputi(0, 0);
	s1.outputpoint();
	s1.area();
	system("pause");
}
```
