```
using namespace std;

class rectangle
{
private:
	int left;
	int top;
	int right;
	int bottom;
public:
	rectangle(int x,int y,int i,int j);
	rectangle(const rectangle &s);

};
```

## 主文件
```
#include "learn.h"

rectangle::rectangle(int x, int y, int i, int j)
{
	left = x;
	top = y;
	right = i;
	bottom = j;
}
rectangle::rectangle(const rectangle &s)
{
	this->left = 0;
	this-> top = 0;
	this->right = s.right;
	this->bottom = s.bottom;
}
void rectangle::show()
{
	cout << "左上角坐标为" << left << "," << top << endl;
	cout << "右下角坐标为" << right << "," << bottom << endl;
}
void rectangle::Diagonal()
{
	
	cout << "对角线长度为" << sqrt((right - left)*(right - left) + (top - bottom)*(top - bottom)) << endl;
}

void main()
{
	rectangle r1(10, 10, 20, 20);
	rectangle r2(r1);
	r1.show();
	r1.Diagonal();
	cout << "===============================" << endl;
	r2.show();
	r2.Diagonal();
	system("pause");
}
```
