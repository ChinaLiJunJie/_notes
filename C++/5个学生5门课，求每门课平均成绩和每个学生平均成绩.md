#include <iostream>
using namespace std;

void main()
{
	float score[5][5];
	float lesson[5];
	float student[5];
	for (int i = 0; i < 5; i++)
	{
		cout << "输入第 " << i+1 << " 个学生的 " << endl;
		for (int j = 0; j < 5; j++)
		{
			cout << "第 " << j+1 << " 门课的成绩 ";
			cin >> score[i][j];
		}
	}
	//每门课的平均成绩
	for (int j = 0; j < 5; j++)
	{
		int sumlesson = 0;
		for (int i = 0; i < 5; i++)
		{
			sumlesson += score[i][j];
		}
		lesson[j] = sumlesson / 5;
	}
	for (int i = 0; i < 5; i++)
		{
			cout << "第" << i + 1 << "门课的平均成绩为 ";
			cout << lesson[i] << endl;
		}

	//每个学生的平均成绩
	for (int i = 0; i < 5; i++)
	{
		int sumlesson = 0;
		for (int j = 0; j < 5; j++)
		{
			sumlesson += score[i][j];
		}
		student[i] = sumlesson / 5;
	}
	for (int i = 0; i < 5; i++)
	{
		cout << "第" << i + 1 << "个学生的平均成绩为 ";
		cout << student[i] << endl;
	}

	system("pause");
}
