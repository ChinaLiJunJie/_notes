# 输入整数，输出倒序数字
```
#include <iostream>
#include <math.h>
using namespace std;

void main()
{
 	int a, 0;//a为输入数字，b为输出数字
	cin >> a;
	int i = a;//i为a的替身
	int c = 0;//c为位数
	//计算输入数字的位数
	while (i != 0)
	{
		c++;
		i /= 10;
	}
	//计算出新数
	while (a != 0)
	{
		int d = a % 10;
		a = a / 10;
		c--;
		b = b +d * pow(10, c);
	}
	cout << b << endl;
	system("pause");
}

#include <iostream>
using namespace std;

void main()
{
	int x;
	int y = 0;
	int z = 0;
	cin >> x;
	while (x != 0)
	{
		y = x % 10;
		z = z * 10 + y;
		x = x / 10;
	}
	cout << z;
	system("pause");
}
```
