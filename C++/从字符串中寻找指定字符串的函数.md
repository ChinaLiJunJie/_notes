#include "stdafx.h"
#include "iostream"
using namespace std;

int search(char *s1, char *s2)
{
	for (unsigned int i = 0; i < strlen(s1); i++)
	{
		if (*(s1 + i) == *s2)
		{
			if (strlen(s2) == 1)
			{
				return 1;
			}
			else {
				int temp = 0;
				for (unsigned j = 1; j < strlen(s1); j++)
				{
					if (*(s1 + i + j) == *(s2 + j))
					{
						temp++;
					}
					else {
						break;
					}
					if (temp == (strlen(s2) - 1))
					{
						return 1;
					}
				}
			}
		}
	}
}

int main()
{
	char fale[] = "abcdefghijklmnopqrstuvwxyz";
	char sear[] = "q";
	if (search(fale, sear) == 1)
	{
		cout << "找到" << endl;
	}
	else {
		cout << "未找到" << endl;
	}
	system("pause");
    return 0;
}
