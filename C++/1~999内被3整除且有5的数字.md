#include <iostream>
using namespace std;

int i = 0;

int choose3(int a)
{
	if (a % 3 == 0)
	{
		return 1;
	}
	else{
		return 0;
	}
}

int choose5(int b)
{
	if (b % 5 == 0 && b % 10 !=0)
	{
		return 1;
	}
	else {
		if (b / 10 % 5 == 0 && b / 10 % 10 != 0)
			return 1;
		else{
			if (b / 10 / 10 % 5 == 0 && b / 10 / 10 % 10 != 0)
				return 1;
			else{
				return 0;
			}
		}
	}
}

void main()
{
	do{
		i += 1;
		if (choose3(i) && choose5(i) == 1)
		{
			cout << i << endl;
		}
	} while (i <= 999);
	system("pause");
}
