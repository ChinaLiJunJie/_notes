//若数组的长度为10.
#include <iostream>
using namespace std;

void main()
{
	int nums[10];
	int max, min,maxi,mini,old;
	for (int i = 0; i < 10; i++)
	{
		cout << "请输入数组第 " << i << " 个元素,从0开始计数，如0，1，2，…" << endl;
		cin >> nums[i];
	}
	//进行最大值的交换
	max = nums[0];
	maxi = 0;
	for (int i = 0; i < 10; i++)
	{
		if ( nums[i] > max )
		{
			max = nums[i];
			maxi = i;
		}
	}
	old = nums[0];
	nums[0] = max;
	nums[maxi] = old;
	
	//进行最小值的交换
	min = nums[0];
	mini = 0;
	for (int i = 0; i < 10; i++)
	{
		if (nums[i] < min)
		{
			min = nums[i];
			mini = i;
		}
	}
	old = nums[9];
	nums[9] = min;
	nums[mini] = old;
	
	cout << "将最大的与第一个元素交换，最小的与最后一个元素交换后数组为" << endl;
	for (int i = 0; i < 10; i++)
	{
		cout << nums[i] << endl;
	}
	system("pause");
}
