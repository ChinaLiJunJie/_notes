## 排序算法
```
#include<iostream>
using namespace std;

void main()
{
	int a[10] = { 8, 45, 38, 28, 97, 53, 27, 63, 18, 66 };
	//插入法
	for (int i = 1; i < 10; i++)
	{
		if (a[i - 1] > a[i])
		{
			int temp = a[i];
			int j = i;
			while (j > 0 && a[j - 1] > temp)
			{
				a[j] = a[j - 1];
				j--;
			}
			a[j] = temp;
		}
	}
	//选择法
	for (int i = 1; i<10; i++)
	{
		for (int j = 0; j<10 - i; j++)
		{
			if (a[j]>a[j + 1])
			{
				int temp = a[j];
				a[j] = a[j + 1];
				a[j + 1] = temp;
			}
		}
	}
	//冒泡法
	for (int i = 0; i<9; i++)
	{
		for (int j = i + 1; j<10; j++)
		{
			if (a[i]<a[j])
			{
				int temp = a[i];
				a[i] = a[j];
				a[j] = temp;
			}
		}
	}
	//打印
	for (int i = 0; i < 10; i++)
	{
		cout << a[i] << endl;
	}
	system("pause");
}
```
