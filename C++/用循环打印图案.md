## 贴边三角
```
#include <iostream>
using namespace std;
//1
void main()
{
	for (int a = 0; a < 6; a++)
	{
		for (int b = 0; b < a*2+1; b ++)
		{
			cout << "*";
		}
		cout << endl;
	}
	system("pause");
}
```
## 中心三角
```
#include <iostream>
using namespace std;

void main()
{
	for (int a = 0; a < 6; a++)
	{
		for (int c = a; c < 6; c++)
		{
			cout << " ";
		}
		for (int b = 0; b < a*2+1; b ++)
		{
			cout << "*";
		}
		cout << endl;
	}
	system("pause");
}
```
## 空心三角
```
#include <iostream>
using namespace std;

void main()
{
	for (int a = 0; a < 6; a++)
	{
		for (int c = a; c < 6; c++)
		{
			cout << " ";
		}
		for (int b = 0; b < a*2+1; b ++)
		{
			if (b==0||b==a*2||a==5)
			{
				cout << "*";
			}
			else{
				cout << " ";
			}
		}
		cout << endl;
	}
	system("pause");
}
```
