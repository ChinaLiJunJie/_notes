# 简易学生查询系统

```
#include <iostream>
using namespace std;

struct stu{
	int id;
	char sex;
	int age;
	float score;
};

void main()
{
	int temp;
	int size;
	cout << "请输入学生个数" << endl;
	cin >> size;
	stu *s = new stu[size];
	for (int i = 0; i < size; i++)
	{
		cout << "正在输入第" << i << "个学生" << endl;
		cout << "请输入学生id" << endl;
		cin >> s[i].id;
		cout << "请输入学生性别" << endl;
		cin >> s[i].sex;
		cout << "请输入学生年龄" << endl;
		cin >> s[i].age;
		cout << "请输入学生成绩" << endl;
		cin >> s[i].score;
	}
	cout << "请输入查询的id" << endl;
	cin >> temp;
	for (int i = 0; i < 3; i++)
	{
		if (s[i].id == temp)
		{
			cout << s[i].id << endl;
			cout << s[i].sex << endl;
			cout << s[i].age << endl;
			cout << s[i].score << endl;
			break;
		}
		if (i == 2)
		{
			cout << "没有找到该学生" << endl;
		}
	}
	system("pause");
}
```
