#include <iostream>
using namespace std;

int sum = 1;

int digui(int i)
{
	sum *= i;
	i--;
	if (i < 1)
	{
		return sum;
	}
	else{
		digui(i);
	}
}

void main()
{
	cout << digui(5) << endl;
	system("pause");
}
