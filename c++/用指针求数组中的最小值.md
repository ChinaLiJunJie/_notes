#include <iostream>
using namespace std;

int *min(int *array, int len);

void main()
{
	int array[] = { 333, 22, 400, 11, 38 };
	int *pMin = NULL;
	pMin = min(array, 5);
	cout << "min = " << *pMin << endl;
	system("pause");
}

int *min(int *array, int len)
{
	static int *p = array;
	for (int i = 1; i < len; i++)
	{
		if (*p > *(array + i))
		{
			p = array + i;
		}
	}
	return p;
}
